function render(data) { let sums = {inimkesksus: 0, elulisus: 0, arusaadavus: 0, loimitus: 0, tegevusele_suunatus: 0}; let count = data.length || 1; let tableHTML = ""; data.forEach(row => { // Konverteerime andmed massiiviks, et pääseda ligi veergudele nende järjekorra järgi (indeksi järgi) // Oletame järjekorda vastavalt sinu tabelile: // 0:date, 1:source, 2:title, 3:url, 4:inimkesksus, 5:elulisus, 6:arusaadavus, 7:loimitus, 8:tegevusele_suunatus, 9:katkestus, 10:tolge const r = Object.values(row); sums.inimkesksus += Number(r[4]) || 0; sums.elulisus += Number(r[5]) || 0; sums.arusaadavus += Number(r[6]) || 0; sums.loimitus += Number(r[7]) || 0; sums.tegevusele_suunatus += Number(r[8]) || 0; tableHTML += ` ${r[2] || 'Pealkiri puudub'}
${r[1] || 'Teadmata'}
"${r[10] || 'AI ootab hindamist...'}" `; }); document.getElementById('score-inim').innerText = Math.round(sums.inimkesksus / count); document.getElementById('score-elul').innerText = Math.round(sums.elulisus / count); document.getElementById('score-arus').innerText = Math.round(sums.arusaadavus / count); document.getElementById('score-loim').innerText = Math.round(sums.loimitus / count); document.getElementById('score-tegev').innerText = Math.round(sums.tegevusele_suunatus / count); document.getElementById('table-body').innerHTML = tableHTML; // ... ülejäänud renderdamise kood (Chart.js osa jääb samaks) } ``` ### 2.
